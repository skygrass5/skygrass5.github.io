迷宫最短路问题

প্রকাশিত হয়েছে 2023-03-19

import collections
from collections import deque
n = int(input())
g = []
dist = [[-1]*n for i in range(n)]#距离数组
for i in range(n):
m = input()
g.append(m)
x1,x2,y1,y2 = map(int,input().split())
dx = [-1,0,1,0]
dy = [0,1,0,-1]
def bfs(maze,start,end):
queue = deque()
queue.append(start)#入队
visited = set()
visited.add(start)
distance = {start:0}
while queue:
cur = queue.popleft()
if cur == end:
return distance[end]
i,j = cur[0],cur[1]
neighbors = [(i-1,j),(i+1,j),(i,j-1),(i,j+1)]
for neighbor in neighbors:
if 0 <= neighbor[0] < len(maze) and 0 <= neighbor[1]<len(maze[0]):
if neighbor not in visited and maze[neighbor[0]][neighbor[1]] != “1”:
queue.append(neighbor)
visited.add(neighbor)
distance[neighbor] = distance[cur]+1

distance = bfs(g,(x1-1,x2-1),(y1-1,y2-1))
print(distance)
